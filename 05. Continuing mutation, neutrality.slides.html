from scipy.stats import bernoulli
class organism:
    """An example of a simple organism"""
    def __init__(self,given_genome=[]):
        """Initialize a genome with a random collection of bits"""
        if len(given_genome) == 0:
            self.genome = bernoulli.rvs(.5,size=30)
        else:
            self.genome = given_genome
    def replicate(self, u = 0):
        """make a copy of a genome, possibly with error"""
        errors = bernoulli.rvs(u,size=len(self.genome))
        new_genome = []
        for i in range(len(self.genome)):
            if errors[i]:
                new_genome.append(int(not self.genome[i]))
            else:
                new_genome.append(int(self.genome[i]))
        return organism(new_genome)

from random import randint
from __future__ import division
def drift(popsize = 10,numgen = 100,startM = 1):  
    # inspired by from http://blog.drewsday.com/2012/08/a-simple-genetic-drift-simulation.html
    """ takes a fixe population size, the number of generations, and a number of mutants.
    Simulates a population sampled with replacement \
    Returns mutant frequencies over the generations"""
    freqM = [startM/popsize]
    prevgen = [1]*startM + [0]*(popsize - startM)
    for gen in range(numgen):
        nextgen = []
        for ind in range(popsize):
            nextgen = nextgen + [prevgen[(randint(0,popsize-1))]]
        freqM.append(sum(nextgen)/popsize)
        prevgen = nextgen
    return(freqM)
